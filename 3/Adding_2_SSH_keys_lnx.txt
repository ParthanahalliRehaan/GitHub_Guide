# ✅ Step 1: Generate SSH key with custom filename
# Mistake made: Thought the file must already exist. 
# Correction: ssh-keygen will create the file automatically.
ssh-keygen -t rsa -C "rehaanparthanahalli@gmail.com"
# When prompted "Enter file in which to save the key (/home/rehaa/.ssh/id_rsa):"
# Type: /home/rehaa/.ssh/id_rsa_not_account
# This creates:
#   ~/.ssh/id_rsa_not_account        (private key)
#   ~/.ssh/id_rsa_not_account.pub    (public key)

# ✅ Step 2: Create SSH config file
# Mistake made: Used Windows-style paths (C:\Users\...) inside WSL/Linux.
# Correction: Use Linux paths (~/.ssh/...).
nano ~/.ssh/config

# ✅ Step 3: Add GitHub host entries
# Mistake made: Initially wrote config with Windows paths.
# Correction: Use proper Linux paths and forward slashes.
# Example config:
Host github-account
    HostName github.com
    User git
    IdentityFile ~/.ssh/id_rsa

Host github-not-account
    HostName github.com
    User git
    IdentityFile ~/.ssh/id_rsa_not_account

# ✅ Step 4: Fix permissions (SSH is strict about this)
chmod 600 ~/.ssh/config
chmod 700 ~/.ssh

# ✅ Step 5: Test connections
# Mistake made: Tried to use plain ssh github.com (wouldn’t pick custom key).
# Correction: Use the alias defined in config.
ssh -T github-account
ssh -T github-not-account
# Expected output: "Hi <username>! You've successfully authenticated..."

# ✅ Step 6: Use aliases in Git remotes
# Mistake made: Using git@github.com directly (ignores config aliases).
# Correction: Use git@<alias>: instead.
git clone git@github-account:username/repo.git
git clone git@github-not-account:username/repo.git

# ✅ Step 7 (Optional): Add keys to SSH agent for convenience
# Mistake made: Forgetting to load keys, requiring -i flag manually.
# Correction: Add keys once per session (or automate in ~/.bashrc).
ssh-add ~/.ssh/id_rsa
ssh-add ~/.ssh/id_rsa_not_account
